Modal

A modal is a dialog box/popup window that is displayed on top of the current page.

HTML


        <div class="modal-display-container displayNone">
            <div class="modal-container">
                <h1>Modal Heading</h1>
                <p>Lorem ipsum dolor sit amet consectetur adipisicing elit.
                Nulla sit impedit exercitationem similique labore dolorum mollitia nam, quibusdam facere ex!</p>
                <div class="flex justify-end">
                    <button class="secondary-btn modal-close-btn modal-close">Reject</button>
                    <button class="primary-outline-btn modal-done-btn">Accept</button>
                </div>
            </div>
        </div>

JavaScript


        const modalOpen = document.querySelector(".modal-demo-btn")
        const modalClose = document.querySelector(".modal-close")
        const modalContainer = document.querySelector(".modal-display-container")

        modalOpen.addEventListener("click",()=>{
            modalContainer.classList.remove("displayNone");
        })
        modalClose.addEventListener("click",()=>{
            modalContainer.classList.add("displayNone");
        })
        
        window.addEventListener("click", event => {
            if (event.target === modalContainer) {
                modalContainer.classList.add("displayNone");
            } 
        });
